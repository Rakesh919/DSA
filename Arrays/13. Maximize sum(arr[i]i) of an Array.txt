class Solution{
    
    int Maximize(int arr[], int n)
    {
        // Complete the function
        Arrays.sort(arr);
       long sum = 0;
       for(int i=0;i<arr.length;i++){
           sum = sum + (long)arr[i]*i;
       }
       return (int)(sum%(1000000007));
    }   
}
